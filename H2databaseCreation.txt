USE idde;

DROP TABLE COMMENTS;
DROP TABLE APARTMENTADS;

CREATE TABLE APARTMENTADS(DESCRIPTION VARCHAR(255), LOCATION VARCHAR(255), NUMBEROFBATHROOMS INT, NUMBEROFROOMS INT, PRICE BIGINT, ID BIGINT(1) primary key);
CREATE TABLE COMMENTS(APARTMENTADID BIGINT, NAME VARCHAR(255), TEXT VARCHAR(255), ID BIGINT PRIMARY KEY, foreign key(APARTMENTADID) references APARTMENTADS(ID) );
